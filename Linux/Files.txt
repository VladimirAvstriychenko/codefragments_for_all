Перемещение
mv notes ../play - перемещаем 'notes' из текущего рабочего каталога в каталог ../play'
mv /usr/tmp/notes . - перемещаем '/usr/tmp/notes' в текущий рабочий каталог
mv work play - перемещаем 'work' из текущего рабочего каталога в каталог 'play'
mv notes notes.old - переименовать файл 'notes' в 'notes.old'
